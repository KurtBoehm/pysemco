parallel Bakery {
  const int P = 8;
  int num[P] = {0[P]}; bool choosing[P] = {false[P]};
  process Process [int i: {0, ..., P - 1}] {
    while (true) {
      choosing[i] = true; int mine = 0;
      for (int k = 0; k < P; ++k) mine = max(mine, num[k]);
      num[i] = mine + 1; choosing[i] = false;
      for (int k = 0; k < P; ++k) {
        while (choosing[k]);
        while (num[k] != 0 && (num[k] < num[i] || (num[k] == num[i] && k < i)));
      }
      // Critical section...
      num[i] = 0;
    }
  }
}
